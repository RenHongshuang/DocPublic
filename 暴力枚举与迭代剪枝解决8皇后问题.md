```
	int ret = 0;
	for(int i=0 ;i<N;i++){
		ret = ret*Base +data[i];

/**
num ->int数

void IToA(int Num,int Base,int Len){
```

## 3. 全遍历与全排列

```
// 全遍历AllTraversal 使用一个N个元素的数组，把N个0到N个M间的所有状态都输出一遍
#include <iostream>
using namespace std;

#define SIZE 10
int data[SIZE];

void AllTraversal(int step,int len,int maxNum){
	// 出口
	if(step == len){
		for (int i=0;i<len ;i++){
			cout<< data[i] << " ";
		}
		cout<< endl;
		return;
	}

	// 迭代遍历每一位
	for (int i =0 ;i <= maxNum;i++){
		data[step] = i;
		AllTraversal(step+1, len, maxNum);
	}
}

int main(){
	//输出 0000---9999
	AllTraversal(0,4,9);

	return 0;
}
```

## 4.皇后问题的两种解法

### 4.1 全遍历思想

```
// 八皇后问题，求8个皇后上下左右对角线方向互不干扰，一共有多少种解法
// 解题思路：
// 如何搜索？用全遍历思想，暴力枚举出所有种棋盘摆法，然后统计符合要求的棋盘数量
// 如何判断安全？通过坐标 x1==x2 || y1 ==y2 || x1-x2 ==y1-y2 || x1-x2 == y2 -y1
// 技巧：判断安全的时候，判断放这枚皇后之前已经放好的皇后是否安全即可
// 技巧：将棋盘的摆法用数组表示 data[SIZE]，其中下标对应横坐标,value对应纵坐标

// 推演：
/*

//同一行只能放一个皇后，所有摆法一共有8的8方个/

# include <iostream>

# define SIZE 8

int data[SIZE];

int isAllSafe(){
	for(int i =1;i<SIZE;i++){

			}
		}
	}
	return true;
}
```

### 4.2 全排列思想

```
// 八皇后问题，求8个皇后上下左右对角线方向互不干扰，一共有多少种解法
// 解题思路：
// 如何搜索？用全遍历思想，暴力枚举出所有种棋盘摆法，然后统计符合要求的棋盘数量
// 如何判断安全？通过坐标 x1==x2 || y1 ==y2 || x1-x2 ==y1-y2 || x1-x2 == y2 -y1
// 技巧：判断安全的时候，判断放这枚皇后之前已经放好的皇后是否安全即可
// 技巧：将棋盘的摆法用数组表示 data[SIZE]，其中下标对应横坐标,value对应纵坐标


#include <iostream>
using namespace std;

#define SIZE 8
int data[SIZE];

// x1 = row, y1 = col
// x2 = i,   y2=data[i]
int isSafe(int row,int col){
	for(int i =0;i<row;i++){
		if(col == data[i] || (row-i == col-data[i]) || (row-i == data[i]-col)){
			return false;	
		}
	}
	return true;
}

int ret =0;
void Allpermutation(int row ){
	if(row == SIZE){
		ret++;
		return;
	}
	for(int i =0;i < SIZE;i++){
		if(isSafe(row,i)){
			data[row] = i;
			Allpermutation(row+1);
		}	
	}
}

int main(){
	Allpermutation(0);
	cout<< ret <<endl;
	return 0;
}
```
